CREATE TABLE PARKING_TB (
    RECORDID VARCHAR2(20) PRIMARY KEY,
    VEHICLENUMBER VARCHAR2(20) NOT NULL,
    SLOTNUMBER VARCHAR2(10) NOT NULL,
    ENTRY_DATETIME TIMESTAMP NOT NULL,
    EXIT_DATETIME TIMESTAMP,
    CHARGES NUMBER,
    REMARKS VARCHAR2(100)
);

CREATE SEQUENCE PARKING_SEQ
START WITH 1
INCREMENT BY 1;
